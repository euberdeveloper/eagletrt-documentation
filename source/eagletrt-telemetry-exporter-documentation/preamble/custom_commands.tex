\newcommand{\headImage}
{
    \begin{tikzpicture}[remember picture,overlay]
        \draw  let \p1=($(current page.north)-(current page header area.south)$),
        \n1={veclen(\x1,\y1)} in
        node [inner sep=0,outer sep=0,below right,yshift=-0.5cm, xshift=-0.5cm] at (current page.north west){\includegraphics[width=\paperwidth]{images/logoEagleHead.pdf}};
    \end{tikzpicture}
}

\newcommand{\footImage}
{
    \begin{tikzpicture}[remember picture,overlay]
        \draw  let \p1=($(current page.south)-(current page header area.south)$),
        \n1={veclen(\x1,\y1)} in
        node [inner sep=0,outer sep=0,below right,yshift=2.3cm] at (current page.south west){\includegraphics[width=\paperwidth]{images/footer.pdf}};
    \end{tikzpicture}
}

\fancypagestyle{plain}{
    \renewcommand{\headrulewidth}{0pt}
    \setlength{\headheight}{15pt}
    \fancyhf{}
    
    \lhead{
        \headImage
    }
    \lfoot{
        \footImage
    }
    \rfoot{Page \thepage}
}
